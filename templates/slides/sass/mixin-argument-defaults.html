Mixin Argument Defaults 


    @mixin foo($width) {
    width: $width;
    }
    @mixin bar($color: blue) {
    color: $color;
    }

    .aMixed {
    @include foo;
    @include bar;
    }
    .anotherMixed {
    @include foo(200px);
    @include bar(red);
    }

outputs


    .aMixed {
    color: blue;
    }
    .anotherMixed {
    width: 200px;
    color: red;
    }
