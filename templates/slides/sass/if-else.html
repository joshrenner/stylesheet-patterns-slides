@if, @else if, @else


		$fancy = 0;

		.foo {
			@if $fancy == 1 {
				border-style: solid;
			}
			@else if $fancy == 2 {
				border-style: dotted;
			}
			@else if $fancy == 3 {
				border-style: double;
			}
			@else {
				border-style: none;
			}
		}

outputs


		.foo {
			border-style: #none;
		}
